1. create virtual environment
	python -m venv venv

2. activate environment
	venv\Scripts\activate.bat

2b. deactivate environmnet
	deactivate

3. install all the packages from requirements
	pip list
	pip install -r requirements.txt

4. create new django project
	django-admin startproject PROJECTNAME

5. create new django app
	python manage.py start app APPNAME

6. add all apps to settings.py

7. run django server
	python manage.py runserver

8. create templates and static folders in project root
9. add in settings:
	'DIRS': [os.path.join(BASE_DIR, 'templates')],
	
	STATICFILES_DIRS = [
    	os.path.join(BASE_DIR, "static"),
	]

10. add media folder in project root
11. add in settings:
	MEDIA_URL = '/media/'

	MEDIA_ROOT = os.path.join(BASE_DIR, 'media')

12. add in root urls
	from django.conf.urls.static import static
	from django.conf import settings

	urlpatterns += static(settings.MEDIA_URL, document_root=settings.MEDIA_ROOT)


13. create templates and static folders in each app 

14. create databases in models.py in each app
	class Project(models.Model):
	    title = models.CharField(max_length=100)

15. register them in the admin.py in each app
	from .models import Project
	admin.site.register(Project)

16. make migrations
	python manage.py makemigrations
	python manage.py migrate

17. map url links in urls.py in root file 
		rom django.urls import path, include

		urlpatterns = [
		    path('admin/', admin.site.urls),
		    path('', include('portfolio_app.urls')),
		    path('blog/', include('blog_app.urls')),
		]

18. create an url file in each app
		from django.urls import path
		from . import views

		app_name = 'blog_app'
		
		urlpatterns = [
		    path('', views.all_blogs, name='all_blogs'),
		]
